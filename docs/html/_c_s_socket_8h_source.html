319 // Try and bind our socket to the provided endpoint.
320 inline int32_t Socket_Bind(Socket* s, IPEndPoint ep)
321 {
322  if (!_cs_g_initialized)
323  {
324  Debug(fputs("CS_Sockets not initialized.\n", stderr));
325  return CS_SOCKET_ERROR;
326  }
327 
328  s->local_ep = ep;
329 
330  void* server_addr = NULL;
331  size_t addr_size = 0;
332 
333  // Resolve the endpoint.
334  switch (ep.address.type)
335  {
336  default:
337  case IPAddressType_IPv4LPStr:
338  s->local_ep.address.ipv4_addr.sin_addr.s_addr = s->local_ep.address.type;
339  s->local_ep.address.ipv4_addr.sin_port = htons(s->local_ep.port);
340  s->local_ep.address.ipv4_addr.sin_family = s->local_ep.addressFamily;
341  server_addr = (void*)&s->local_ep.address.ipv4_addr;
342  addr_size = sizeof(s->local_ep.address.ipv4_addr);
343  break;
344  }
345 
346  // Try and bind.
347  const int32_t bind_res = bind(s->_native_handle, (struct sockaddr*)server_addr, addr_size);
348  if (bind_res == CS_SOCKET_ERROR)

350  Debug(fputs("CS_Sockets: Failed to bind socket.\n", stderr));
351  Debug(perror("native error"));
352  CS_CLOSE_SOCKET(s->_native_handle);
353  return CS_SOCKET_ERROR;
354  }
