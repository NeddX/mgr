169  {
170  const auto ip_endpoint = IPEndPoint_New(net::IPAddress_Parse(Application::RootServerIP),
171  net::AddressFamily_InterNetwork, Application::RootServerPort);
172 
173  if (net::Socket_Connect(m_Socket, ip_endpoint) == CS_SOCKET_ERROR)
174  return Err{ ErrType::NetConnectionTimeout, "Failed to connect to ({}:{}).", ip_endpoint.address.str,
175  m_Ep.port };
176 
177  // Grab Node ID from /etc/vlink.conf.
178  auto node_file = utils::fs::ReadToString("/etc/vlink.conf");
179  if (node_file)
180  m_NodeID = static_cast<u8>(std::stoi(utils::StrSplit(node_file.Unwrap(), '=')[1]));
181  else
182  return node_file.UnwrapErr();
183 
184  m_Logger->Info("Node ID: {}", m_NodeID);
185  m_Logger->Info("Connected to Root Complex.");
186  m_Logger->Info("Sending InitConn packet...");
187 
188  // Send Reply to register as an Endpoint.
189  net::Packet initconn{ net::PacketType::Ready };
190  initconn << m_NodeID;
191  TRY_UNWRAP(net::BeginSend(m_Socket, std::move(initconn)));
192 
193  // Wait for InitCon acknowledgement.
194  if (const auto result = net::BeginReceive(m_Socket); !result || result.Unwrap().Type() != net::PacketType::Ok)
195  return Err{ ErrType::NetInitConFailure };
196 
197  // Request for configuration.
198  if (m_CrewStation)
199  {
200  TRY_UNWRAP(net::BeginSend(m_Socket, net::PacketType::GetCrewConfig));
201  auto result = net::BeginReceive(m_Socket);
202  if (!result)
203  return result.UnwrapErr();
204 
205  // Check if the packet itself is an error.
206  auto the_horror = result.Unwrap();
207  if (!the_horror)
208  return Err::FromPacket(std::move(the_horror));
209 
210  // Parse the configuration.
211  std::string jsonstr;
212  result.Unwrap() >> jsonstr;
213 
214  m_Logger->Log(__func__, lgx::Level::Info, "Crew config: {}", jsonstr);
215  }
216  else if (m_Concentrator)
217  {
218  TRY_UNWRAP(net::BeginSend(m_Socket, net::PacketType::GetCtrConfig));
219  auto result = net::BeginReceive(m_Socket);
220  if (!result)
221  return result.UnwrapErr();
222 
223  // Check if the packet itself is an error.
224  auto the_horror = result.Unwrap();
225  if (!the_horror)
226  return Err::FromPacket(std::move(the_horror));
227 
228  // Parse the configuration.

230  result.Unwrap() >> jsonstr;
231 
232  // Convert to json and deserialise it from json to vector of cameras.
233  nlohmann::json j = nlohmann::json::parse(jsonstr);

235 
236  // Validate cameras.

241  }
